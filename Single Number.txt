
class Solution {
    public int singleNumber(int[] nums) {
        Arrays.sort(nums);
        int sol=0;
        for (int i=0; i<nums.length; i++){
            if (i == nums.length-1){
                sol = nums[i];
            }
            else if (nums[i] == nums[i+1]){
                i++;
            }
            else{
                sol = nums[i];
            }
        }
    return sol;
    } 
}
